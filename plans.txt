    the machine chains
    
        VF Gauntlet     :: turns arbitrary items into internal energy
                            by default, 1 item = 2 energy
                            for every 1 energy, produces one "mundane matter"
                            -- input top; output bottom

        VF Calcinator   :: turns "mundane matter" into internal energy
                            by default, 1 matter = 3 energy
                            for every 1 energy, produces one "crude matter"
                            -- requires fuel (anything a furnace can burn)
                            -- input top; output bottom

            [future directions: an advanced calcinator that burns energetic matter for improved yield?]

        VF Sieve        :: turns "crude matter" into internal energy
                            by default, 1 matter = 3 energy
                            for every 1 energy, produces one "raw matter"
                            -- input top; output bottom

        VF Press        :: turns "raw matter" into internal energy
                            by default, 1 matter = 0.005 energy
                            for every 1 energy, produces one "compacted matter"
                            -- input top; output bottom

        VF Kiln         :: turns 1 "compacted matter" into 1 diamond
                            -- must be in 'low power mode'
                            -- requires at least one fueled Etna
                            -- input sides; output top or bottom

        ----------

        VF Gauntlet     :: turns 1 diamond into internal gem-energy
                            by default, 1 diamond = 1 gem-energy
                            for every 1 gem-energy, produces one "gem matter"

        VF Press        :: turns 1 energetic matter into internal subvorpal-energy
                            by default, 1 matter = 0.334 subvorpal-energy
                            for every 1 subvorpal-energy, produces one "subvorpal matter"

        ----------

        VF Rectifier    :: turns subvorpal matter into subvorpal ingot, or vorpal matter into vorpal ingot
                            -- uses energetic matter as fuel
                            -- input top; output sides

        ----------

        VF Kiln         :: turns subvorpal matter into internal energy
                            by default, 1 matter = 0.334 energy
                            for every 1 energy, produces one "vorpal matter"
                            -- must be in 'high power mode'
                            -- requires three fueled Etnas

        VF Etna         :: attaches to and heats a Kiln
                            -- uses energetic matter as fuel

        ----------

        VF Liquefactor  :: turns 1 "vorpal matter" into internal energy
                            also stores water
                            for every 1 energy + 100mB water, produces 100mB "vorpal slurry"
                            -- [inputs/outputs?]
                            -- will automatically pull water from an adjacent fluid inventory
                            -- [automatically output to adjacent empty or slurry-containing fluid inventory?]

        VF Smithy       :: holds up to 1 VF Hammer
                            also stores "vorpal slurry"
                            used for crafting advanced items
                            -- [any kind of automation options?]
                            -- will automatically pull slurry from an adjacent fluid inventory

        -- Vorpal Matter [Ingot]
            Gauntlet(diamond) -> Energetic Matter -> Press -> Kiln/Etna [-> Rectifier]

        -- Subvorpal Matter [Ingot]
            Gauntlet(diamond) -> Energetic Matter -> Press [-> Rectifier]

        -- Diamond
            Gauntlet -> Calcinator -> Sieve -> Press -> Kiln
